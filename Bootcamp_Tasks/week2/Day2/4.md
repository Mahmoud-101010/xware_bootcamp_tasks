
##CREATE TABLE student 
```MYSQL
    (
    id INT PRIMARY KEY NOT NULL,
    name VARCHAR(20) NOT NULL,
    age INT NOT NULL
    );
```



##CREATE TABLE course
```MSQL 
    (

    name VARCHAR(20) NOT NULL
    );
```

```MSQL
CREATE TABLE enrolment (
    id INT PRIMARY KEY NOT NULL,

);
```
